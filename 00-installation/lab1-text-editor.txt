## VIM

- Edit Mode
- Command Mode
- Visual Mode

- i <- edit mode
- esc <- exit from edit mode
- :q <- quit
- :wq <- save and quit
- :q! <- force quit
- yy <- copy line
- 5yy <- copy 5 lines
- p <- paste
- dd <- delete line
- 3dd <- delete 3 lines
- ?adit <- search "adit"
- n <- next result
- :%s/text1/text2 <- search and replace text1 to text2

## NANO

- ^K <- cut
- ^U <- uncut
- ^O <- save
- ^X <- exit
